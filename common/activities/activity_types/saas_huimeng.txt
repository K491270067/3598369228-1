# Huimeng activity: a diplomatic assembly that creates or expands a vanilla House Bloc.

activity_huimeng = {
	is_shown = { saas_is_ritual_collapse_phase_trigger = yes }

	can_start_showing_failures_only = {
		NOT = { is_activity_type_on_cooldown = activity_huimeng }
		saas_can_host_huimeng_trigger = yes
		is_available_adult = yes
		is_at_war = no
	}

	can_always_plan = no
	is_single_location = yes
	cooldown = { years = 10 }
	province_filter = top_realm
	ai_province_filter = domain
	max_province_icons = 5

	is_location_valid = {
		has_holding = yes
		county.holder ?= {
			OR = {
				this = scope:host
				any_liege_or_above = { this = scope:host }
			}
		}
	}

	is_valid = {
		scope:host = {
			is_alive = yes
			is_incapable = no
			is_imprisoned = no
			saas_can_host_huimeng_trigger = yes
		}
		trigger_if = {
			limit = { is_current_phase_active = yes }
			has_attending_activity_guests = yes
		}
	}

	on_invalidated = {
		
		if = {
			limit = { has_attending_activity_guests = no }
			scope:host = { trigger_event = activity_system.0100 }
		}
	}

	on_host_death = {
		if = { limit = { exists = scope:host.player_heir } scope:host.player_heir = { save_scope_as = new_host } }
		else = { scope:host.primary_heir = { save_scope_as = new_host } }
		set_activity_host = scope:new_host
	}

	phases = {
		huimeng_phase_preparations = {
			location_source = pickable
			is_predefined = yes
			order = 1
			on_phase_active = {
				if = {
					limit = { this = scope:host }
					trigger_event = huimeng_activity.0001
					scope:activity = {
						every_attending_character = {
							limit = { this != scope:host }
							if = {
								limit = {
									OR = {
										has_activity_intent = huimeng_offer_support
										opinion = { target = scope:host value >= 25 }
										is_allied_to = scope:host
									}
								}
								scope:activity = {
									add_to_guest_subset = { name = supporter target = prev }
									activity_special_type_progression_tiny = yes
								}
							}
							else_if = {
								limit = {
									OR = {
										has_activity_intent = huimeng_disrupt_loyalty
										opinion = { target = scope:host value <= -25 }
									}
								}
								scope:activity = {
									add_to_guest_subset = { name = detractor target = prev }
									activity_special_type_progression_variable = { NUM = -6 }
								}
							}
						}
						progress_activity_phase_after = { weeks = 10 }
					}
				}
			}
		}

		huimeng_phase_deliberations = {
			location_source = pickable
			is_predefined = yes
			order = 2
			on_phase_active = {
				if = { limit = { this = scope:host } trigger_event = huimeng_activity.0010 }
				scope:activity = { progress_activity_phase_after = { weeks = 3 } }
			}
		}

		huimeng_phase_oath = {
			location_source = pickable
			is_predefined = yes
			order = 3
			on_phase_active = {
				if = { limit = { this = scope:host } trigger_event = huimeng_activity.0020 }
				scope:activity = { progress_activity_phase_after = { days = 2 } }
			}
		}
	}

	guest_subsets = { supporter detractor }

	on_start = {
		activity_setup_special_type_progression_effect = yes
		# Huimeng begins at the embarrassing tier. Events and guest alignment
		# then change magnificence exactly like vanilla special progression.
		activity_special_type_progression_variable = { NUM = 30 }
	}

	open_invite = yes
	max_guests = { value = 60 }
	reserved_guest_slots = 5
	allow_zero_guest_invites = no

	can_be_activity_guest = {
		in_diplomatic_range = scope:host
		is_available_adult = yes
		is_landed = yes
		highest_held_title_tier >= tier_duchy
		religion = religion:lijiao_religion
	}

	host_intents = {
		intents = { huimeng_display_authority huimeng_isolate_opponents huimeng_unite_supporters }
		default = huimeng_display_authority
	}
	guest_intents = {
		intents = { huimeng_witness_oath huimeng_offer_support huimeng_present_domain huimeng_assert_rights huimeng_seek_advantage huimeng_disrupt_loyalty }
		default = huimeng_witness_oath
	}

	guest_join_chance = {
		base = 10
		base_activity_modifier = yes
		activity_guest_shared_ai_accept_modifier = yes
		activity_guest_health_ai_accept_modifier = yes
		modifier = { add = 25 opinion = { target = scope:host value >= 50 } }
		modifier = { add = -50 opinion = { target = scope:host value <= -50 } }
	}

	guest_invite_rules = {
		rules = {
			1 = activity_invite_rule_vassals
		}
		defaults = {
			4 = activity_invite_neighbouring_rulers
			3 = activity_invite_rule_tributaries
		}
	}

	wait_time_before_start = { months = 4 }
	max_guest_arrival_delay_time = { months = 1 }
	cost = { gold = { value = 200 } }
	ui_predicted_cost = { gold = { value = 200 } }
	ai_will_do = { value = 20 }
	ai_check_interval_by_tier = { barony = 0 county = 0 duchy = 0 kingdom = 60 empire = 80 hegemony = 0 }

	on_complete = {
		if = {
			limit = { this = scope:host }
			if = {
				limit = {
					scope:activity.var:activity_special_type_progression >= 50
					scope:activity.var:activity_special_type_progression < 80
				}
				add_character_modifier = { modifier = saas_huimeng_modest_attraction_modifier years = 5 }
				saas_apply_huimeng_intent_reward_effect = { YEARS = 5 }
				scope:activity = { every_attending_character = { saas_apply_huimeng_intent_reward_effect = { YEARS = 5 } } }
				saas_complete_huimeng_effect = yes
			}
			else_if = {
				limit = {
					scope:activity.var:activity_special_type_progression >= 80
					scope:activity.var:activity_special_type_progression < 95
				}
				add_character_modifier = { modifier = saas_huimeng_grand_attraction_modifier years = 10 }
				saas_apply_huimeng_intent_reward_effect = { YEARS = 10 }
				scope:activity = { every_attending_character = { saas_apply_huimeng_intent_reward_effect = { YEARS = 10 } } }
				saas_complete_huimeng_effect = yes
				if = {
					limit = { saas_is_wangdao_trigger = yes }
					saas_trigger_catalyst_effect = {
						CATALYST = saas_catalyst_successful_huimeng
						CHARACTER = scope:host
					}
				}
			}
			else_if = {
				limit = { scope:activity.var:activity_special_type_progression >= 95 }
				add_character_modifier = { modifier = saas_huimeng_splendid_attraction_modifier years = 15 }
				saas_apply_huimeng_intent_reward_effect = { YEARS = 15 }
				scope:activity = { every_attending_character = { saas_apply_huimeng_intent_reward_effect = { YEARS = 15 } } }
				saas_complete_huimeng_effect = yes
				if = {
					limit = { saas_is_wangdao_trigger = yes }
					saas_trigger_catalyst_effect = {
						CATALYST = saas_catalyst_successful_huimeng
						CHARACTER = scope:host
					}
				}
			}
			else = { trigger_event = huimeng_activity.0090 }
		}
	}

	activity_window_widgets = {
		huimeng_special_type_progression = "activity_plugin_widgets_summary"
	}

	background = {
		texture = "gfx/interface/illustrations/event_scenes/tgp_study_asia.dds"
		environment = "environment_tgp_study_asia"
		ambience = "event:/DLC/EP4/SFX/Events/Event_Backgrounds/tgp_study_asia"
	}
}
