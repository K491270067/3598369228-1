namespace = dm_reform

dm_reform.1101 = {
	type = character_event
	title = dm_reform.1101.t
	desc = dm_reform.1101.desc
	theme = court_event
	left_portrait = scope:dm_reform_actor
	immediate = {
		scope:dm_reform_story = {
			random_in_list = {
				variable = dm_reform_opponents
				save_scope_as = dm_reform_actor
			}
		}
		if = {
			limit = { NOT = { exists = scope:dm_reform_actor } }
			root = { save_scope_as = dm_reform_actor }
		}
	}
	option = {
		name = dm_reform.event.measured_response
		custom_tooltip = dm_reform.event.chance_tt
		random_list = {
			50 = {
				modifier = {
					add = 35
					root = { martial >= 22 }
				}
				modifier = {
					add = 25
					scope:dm_reform_story = {
						dm_reform_has_reformer_trigger = yes
						var:dm_reform_reformer = { martial >= 22 }
					}
				}
				modifier = {
					add = 10
					root = { OR = { has_trait = brave has_trait = strategist } }
				}
				scope:dm_reform_story = {
					change_variable = { name = dm_reform_progress add = 3 }
					change_variable = { name = dm_reform_support add = 1 }
					set_variable = { name = dm_reform_last_result value = flag:success }
				}
			trigger_event = { id = dm_reform.2101 days = 1 }
			}
			50 = {
				scope:dm_reform_story = {
					change_variable = { name = dm_reform_progress add = -4 }
					change_variable = { name = dm_reform_support add = -2 }
					set_variable = { name = dm_reform_last_result value = flag:failure }
				}
			trigger_event = { id = dm_reform.2101 days = 1 }
			}
		}
	}
	option = {
		name = dm_reform.event.costly_response
		trigger = { prestige >= 100 }
		add_prestige = -100
		scope:dm_reform_story = {
			change_variable = { name = dm_reform_progress add = 5 }
			change_variable = { name = dm_reform_support add = 1 }
			set_variable = { name = dm_reform_last_result value = flag:great_success }
		}
			trigger_event = { id = dm_reform.2101 days = 1 }
	}
}

	title = dm_reform.1111.t
	desc = dm_reform.1111.desc
	theme = court_event
	left_portrait = scope:dm_reform_actor
	immediate = {
		scope:dm_reform_story = {
			random_in_list = {
				variable = dm_reform_opponents
				save_scope_as = dm_reform_actor
			}
		}
		if = {
			limit = { NOT = { exists = scope:dm_reform_actor } }
			root = { save_scope_as = dm_reform_actor }
		}
	}
	option = {
		name = dm_reform.event.measured_response
		custom_tooltip = dm_reform.event.chance_tt
		random_list = {
			50 = {
				modifier = {
					add = 35
					root = { intrigue >= 18 }
				}
				modifier = {
					add = 25
					scope:dm_reform_story = {
						dm_reform_has_reformer_trigger = yes
						var:dm_reform_reformer = { intrigue >= 18 }
					}
				}
				modifier = {
					add = 10
					root = { OR = { has_trait = deceitful has_trait = schemer } }
				}
				scope:dm_reform_story = {
					change_variable = { name = dm_reform_progress add = 2 }
					change_variable = { name = dm_reform_support add = 1 }
					set_variable = { name = dm_reform_last_result value = flag:success }
				}
			trigger_event = { id = dm_reform.2111 days = 1 }
			}
			50 = {
				scope:dm_reform_story = {
					change_variable = { name = dm_reform_progress add = -2 }
					change_variable = { name = dm_reform_support add = -1 }
					set_variable = { name = dm_reform_last_result value = flag:failure }
				}
			trigger_event = { id = dm_reform.2111 days = 1 }
			}
		}
	}
	option = {
		name = dm_reform.event.costly_response
		trigger = { prestige >= 100 }
		add_prestige = -100
		scope:dm_reform_story = {
			change_variable = { name = dm_reform_progress add = 4 }
			change_variable = { name = dm_reform_support add = 1 }
			set_variable = { name = dm_reform_last_result value = flag:great_success }
		}
			trigger_event = { id = dm_reform.2111 days = 1 }
	}
}
